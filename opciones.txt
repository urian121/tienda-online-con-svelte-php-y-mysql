import { writable } from "svelte/store";

let URL_API = "http://localhost/tienda-onlne-con-svelte-php-y-mysql/backend-php/index.php";

export const cart = writable([]);

export async function getCart() {
  try {
    const response = await fetch(`${URL_API}?action=getCart`);
    if (!response.ok) {
      throw new Error(`Error en la petici√≥n: ${response.statusText}`);
    }

    const data = await response.json();
    console.log("Carrito recibido del backend:", data); // Verifica los datos en la consola
    cart.set(data);
  } catch (error) {
    console.error("Error al obtener el carrito:", error);
  }
}
